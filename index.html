function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); addHeaderIfNeeded(sheet); var data = JSON.parse(e.postData.contents); appendRow(sheet, data); return buildResponse({result:"success"}, null); } catch(err) { return buildResponse({result:"error", message:err.toString()}, null); } } function doGet(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); addHeaderIfNeeded(sheet); var p = e.parameter; var cb = p.callback || null; if (p.action === "getRecords") { var records = getRecords(sheet, p.업체명 || ""); return buildResponse({result:"success", data: records}, cb); } if (p.action === "getReservations") { var reservations = getReservations(sheet, p.장비명 || ""); return buildResponse({result:"success", data: reservations}, cb); } // 방법 B: 진행중(종료일 미정)인 이용 기록 조회 if (p.action === "getOpenRecords") { var open = getOpenRecords(sheet, p.업체명 || "", p.장비명 || ""); return buildResponse({result:"success", data: open}, cb); } // 방법 B: 진행중 기록을 찾아 같은 행에 종료 정보 업데이트 if (p.action === "closeRecord") { try { closeRecord(sheet, p); return buildResponse({result:"success"}, cb); } catch(closeErr) { return buildResponse({result:"error", message:closeErr.toString()}, cb); } } if (p.업체명) { appendRow(sheet, p); } return buildResponse({result:"success"}, cb); } catch(err) { return buildResponse({result:"error", message:err.toString()}, e.parameter.callback||null); } } function appendRow(sheet, data) { // 사용량 처리: L단위 또는 챔버(마우스/랫)만 표기 var m = parseInt(data.마우스수) || 0; var r = parseInt(data.랫수) || 0; var usageQty = ""; var animalType = ""; if (m > 0 || r > 0) { if (m > 0 && r > 0) { usageQty = m + r; animalType = "마우스 " + m + "마리 / 랫 " + r + "마리"; } else if (m > 0) { usageQty = m; animalType = "마우스"; } else { usageQty = r; animalType = "랫"; } } else if (data.사용량_L) { usageQty = data.사용량_L; } // Hour/Day 단위는 사용량 빈칸 // 날짜/시간 분리 var startDate = data.사용시작일 || data.예약시작일 || ""; var startTime = data.사용시작시간 || ""; var endDate = data.사용종료일 || data.예약종료일 || ""; var endTime = data.사용종료시간 || ""; // 단기 장비: 사용시작/종료는 시간만 있음 if (data.사용시작 && !startTime) startTime = data.사용시작; if (data.사용종료 && !endTime) endTime = data.사용종료; // 사용일수 자동 계산 var usageDays = ""; if (startDate && endDate && startDate !== "미정" && endDate !== "미정") { var s = new Date(startDate); var e2 = new Date(endDate); var diff = Math.round((e2 - s) / 86400000); usageDays = diff >= 0 ? diff + 1 : 1; } else if (data.사용일수 || data.예약일수) { usageDays = data.사용일수 || data.예약일수; } // 휴대폰 앞자리 0 보존 - 텍스트로 강제 설정 var phone = ""; if (data.휴대폰) { phone = String(data.휴대폰).replace(/[^0-9]/g, ""); if (phone.length >= 9 && !phone.startsWith("0")) phone = "0" + phone; } var newRow = sheet.getLastRow() + 1; sheet.appendRow([ data.업체명 || "", data.장비명 || "", startDate, endDate, usageDays, startTime, endTime, usageQty, animalType, data.단가 || data.단가_마우스 || "", data.수수료 || "", data.구분 || "이용", data.담당자 || "", phone, ]); // 서식 적용 (가운데정렬, 맑은고딕 11pt) var range = sheet.getRange(newRow, 1, 1, 14); range.setHorizontalAlignment("center"); range.setVerticalAlignment("middle"); range.setFontFamily("Malgun Gothic"); range.setFontSize(11); // 휴대폰 칸 텍스트 강제 if (phone) { sheet.getRange(newRow, 14).setNumberFormat("@"); sheet.getRange(newRow, 14).setValue(phone); } } // 업체명 느슨한 비교용 정규화 함수 function normalizeCompany(name) { if (!name) return ""; return String(name) .replace(/[㈜㈔]/g, "") .replace(/[（(]\s*[주사]\s*[）)]/g, "") .replace(/주식회사/g, "") .replace(/\s+/g, "") .trim(); } function getRecords(sheet, 업체명) { var lastRow = sheet.getLastRow(); if (lastRow < 2) return []; var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; var data = sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).getValues(); var idx = {}; headers.forEach(function(h, i) { idx[h] = i; }); var target = normalizeCompany(업체명); var records = []; data.forEach(function(row) { var 업체 = String(row[idx["업체명"]] || "").trim(); if (target && normalizeCompany(업체) !== target) return; records.push({ 업체명: 업체, 장비명: String(row[idx["장비명"]] || ""), 사용시작일: String(row[idx["사용시작일"]] || ""), 사용종료일: String(row[idx["사용종료일"]] || ""), 사용일수: String(row[idx["사용일수"]] || ""), 시작시간: String(row[idx["시작시간"]] || ""), 종료시간: String(row[idx["종료시간"]] || ""), 사용량: String(row[idx["사용량"]] || ""), 장비사용료: String(row[idx["장비사용료"]] || ""), 구분: String(row[idx["구분"]] || ""), }); }); return records; } function getReservations(sheet, 장비명) { var lastRow = sheet.getLastRow(); if (lastRow < 2) return []; var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; var data = sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).getValues(); var idx = {}; headers.forEach(function(h, i) { idx[h] = i; }); var reservations = []; data.forEach(function(row) { var 구분Col = idx["구분"] !== undefined ? idx["구분"] : row.length - 1; var 구분 = String(row[구분Col] || "").trim(); if (구분 !== "예약") return; var 장비 = String(row[idx["장비명"]] || "").trim(); if (장비명 && 장비 !== 장비명) return; var startCol = idx["사용시작일"] !== undefined ? idx["사용시작일"] : 2; var endCol = idx["사용종료일"] !== undefined ? idx["사용종료일"] : 3; var s = row[startCol], en = row[endCol]; if (s instanceof Date) s = Utilities.formatDate(s, "Asia/Seoul", "yyyy-MM-dd"); else if (typeof s === "string" && s.indexOf("T") > -1) s = s.split("T")[0]; if (en instanceof Date) en = Utilities.formatDate(en, "Asia/Seoul", "yyyy-MM-dd"); else if (typeof en === "string" && en.indexOf("T") > -1) en = en.split("T")[0]; reservations.push({ 장비명: 장비, 업체명: String(row[idx["업체명"]] || ""), 예약시작일: String(s || ""), 예약종료일: String(en || "") }); }); return reservations; } // 방법 B: 아직 종료되지 않은(사용종료일이 비어있거나 "미정") 이용 건 조회 function getOpenRecords(sheet, 업체명, 장비명) { var lastRow = sheet.getLastRow(); if (lastRow < 2) return []; var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; var data = sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).getValues(); var idx = {}; headers.forEach(function(h, i) { idx[h] = i; }); var target = normalizeCompany(업체명); var results = []; data.forEach(function(row, i) { var 구분 = String(row[idx["구분"]] || "").trim(); if (구분 !== "이용") return; var endVal = row[idx["사용종료일"]]; var endStr = endVal instanceof Date ? Utilities.formatDate(endVal, "Asia/Seoul", "yyyy-MM-dd") : String(endVal || "").trim(); var isOpen = (endStr === "" || endStr === "미정"); if (!isOpen) return; var 업체 = String(row[idx["업체명"]] || "").trim(); if (target && normalizeCompany(업체) !== target) return; var 장비 = String(row[idx["장비명"]] || "").trim(); if (장비명 && 장비 !== 장비명) return; var s = row[idx["사용시작일"]]; if (s instanceof Date) s = Utilities.formatDate(s, "Asia/Seoul", "yyyy-MM-dd"); var startTimeStr = String(row[idx["시작시간"]] || "").trim(); // 시작일/시작시간이 없는 행은 종료 처리할 기준이 없으므로 목록에서 제외 if (!s || !startTimeStr) return; results.push({ rowIndex: i + 2, // 실제 시트 행 번호 (헤더 1행 + 0-index 보정) 업체명: 업체, 장비명: 장비, 사용시작일: String(s || ""), 시작시간: startTimeStr, 담당자: String(row[idx["담당자"]] || ""), 단가: String(row[idx["단가"]] || "") }); }); return results; } // 방법 B: 지정된 행을 찾아 종료 정보로 업데이트 (새 행을 추가하지 않음) function closeRecord(sheet, p) { var rowIndex = parseInt(p.rowIndex); if (!rowIndex || rowIndex < 2) throw new Error("잘못된 행 번호입니다."); var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; var idx = {}; headers.forEach(function(h, i) { idx[h] = i; }); var lastRow = sheet.getLastRow(); if (rowIndex > lastRow) throw new Error("존재하지 않는 행입니다."); var rowVals = sheet.getRange(rowIndex, 1, 1, sheet.getLastColumn()).getValues()[0]; var 구분 = String(rowVals[idx["구분"]] || "").trim(); var 업체 = String(rowVals[idx["업체명"]] || "").trim(); var 장비 = String(rowVals[idx["장비명"]] || "").trim(); // 안전 검증: 요청받은 정보와 실제 행이 일치하는지 재확인 (동시성/오류 방지) if (구분 !== "이용") throw new Error("이미 종료되었거나 이용 건이 아닌 행입니다."); if (p.업체명 && normalizeCompany(업체) !== normalizeCompany(p.업체명)) throw new Error("업체명이 일치하지 않습니다."); if (p.장비명 && 장비 !== p.장비명) throw new Error("장비명이 일치하지 않습니다."); sheet.getRange(rowIndex, idx["사용종료일"] + 1).setValue(p.사용종료일 || ""); sheet.getRange(rowIndex, idx["종료시간"] + 1).setValue(p.종료시간 || ""); sheet.getRange(rowIndex, idx["사용일수"] + 1).setValue(p.사용일수 || ""); sheet.getRange(rowIndex, idx["장비사용료"] + 1).setValue(p.수수료 || ""); var range = sheet.getRange(rowIndex, 1, 1, sheet.getLastColumn()); range.setHorizontalAlignment("center"); range.setVerticalAlignment("middle"); range.setFontFamily("Malgun Gothic"); range.setFontSize(11); } function addHeaderIfNeeded(sheet) { if (sheet.getLastRow() === 0) { sheet.appendRow([ "업체명","장비명", "사용시작일","사용종료일","사용일수", "시작시간","종료시간", "사용량","동물종", "단가","장비사용료","구분", "담당자","휴대폰" ]); var hRange = sheet.getRange(1, 1, 1, 14); hRange.setHorizontalAlignment("center"); hRange.setVerticalAlignment("middle"); hRange.setFontFamily("Malgun Gothic"); hRange.setFontSize(11); hRange.setFontWeight("bold"); } } function formatDateVal(val) { if (!val) return ""; if (val instanceof Date) return Utilities.formatDate(val, "Asia/Seoul", "yyyy-MM-dd"); var s = String(val); if (s.indexOf("GMT") > -1 || s.indexOf("표준시") > -1) { var d = new Date(s); if (!isNaN(d.getTime())) return Utilities.formatDate(d, "Asia/Seoul", "yyyy-MM-dd"); } if (s.indexOf("T") > -1) return s.split("T")[0]; if (/^\d{4}-\d{2}-\d{2}/.test(s)) return s.slice(0,10); return s; } function buildResponse(obj, callback) { var json = JSON.stringify(obj); if (callback) { return ContentService .createTextOutput(callback + "(" + json + ");") .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
